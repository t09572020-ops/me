        let slideTimeout;
        let currentMediaList = [];
        let currentMediaIndex = 0;

        function isVideo(filename) {
            return filename.toLowerCase().match(/\.(mp4|webm|mov|ogg)$/i);
        }

        function openModal(mediaSource, title, description) {
            const modal = document.getElementById('imageModal');
            const backdrop = document.getElementById('modalBackdrop');
            const content = document.getElementById('modalContent');
            const vid = document.getElementById('modalVideo');
            const mTitle = document.getElementById('modalTitle');
            const mDesc = document.getElementById('modalDescription');

            clearTimeout(slideTimeout);
            vid.pause();
            vid.onended = null;

            if (Array.isArray(mediaSource)) {
                currentMediaList = mediaSource;
            } else {
                currentMediaList = [mediaSource];
            }
            currentMediaIndex = 0;

            mTitle.innerText = title;
            mDesc.innerText = description;

            updateMedia();

            modal.classList.remove('hidden');
            setTimeout(() => {
                backdrop.classList.remove('opacity-0');
                content.classList.remove('opacity-0', 'scale-95');
                content.classList.add('opacity-100', 'scale-100');
            }, 10);
        }

        function updateMedia() {
            if (currentMediaList.length === 0) return;
            const src = currentMediaList[currentMediaIndex];
            
            const img = document.getElementById('modalImage');
            const vid = document.getElementById('modalVideo');
            const fallback = document.getElementById('modalImageFallback');
            const prevBtn = document.getElementById('modalPrevBtn');
            const nextBtn = document.getElementById('modalNextBtn');
            
            img.classList.add('hidden');
            vid.classList.add('hidden');
            fallback.classList.add('hidden');

            if (currentMediaList.length > 1) {
                prevBtn.classList.remove('hidden');
                nextBtn.classList.remove('hidden');
            } else {
                prevBtn.classList.add('hidden');
                nextBtn.classList.add('hidden');
            }

            if (isVideo(src)) {
                vid.src = src;
                vid.classList.remove('hidden', 'opacity-0');
                vid.loop = (currentMediaList.length === 1);
                vid.play().catch(e => console.log("影片自動播放可能被阻擋:", e));
                
                if (currentMediaList.length > 1) {
                    vid.onended = () => { fadeAndNext(); };
                }
            } else {
                img.src = src;
                img.classList.remove('hidden', 'opacity-0');
                
                if (currentMediaList.length > 1) {
                    slideTimeout = setTimeout(fadeAndNext, 3000); 
                }
            }

            img.onerror = function() {
                img.classList.add('hidden');
                fallback.classList.remove('hidden');
                if (currentMediaList.length > 1) {
                    slideTimeout = setTimeout(fadeAndNext, 3000);
                }
            };
        }

        function nextMedia(e) {
            if (e) e.stopPropagation();
            clearTimeout(slideTimeout); 
            const img = document.getElementById('modalImage');
            const vid = document.getElementById('modalVideo');
            img.classList.add('opacity-0');
            vid.classList.add('opacity-0');
            setTimeout(() => {
                currentMediaIndex = (currentMediaIndex + 1) % currentMediaList.length;
                updateMedia();
            }, 200);
        }

        function prevMedia(e) {

        async function fetchWithRetry(url, options, maxRetries = 5) {
            const delays = [1000, 2000, 4000, 8000, 16000];
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                    return await response.json();
                } catch (error) {
                    if (i === maxRetries - 1) throw error;
                    await new Promise(res => setTimeout(res, delays[i]));
                }
            }
        }

        function appendMessage(role, content) {
            const messagesDiv = document.getElementById('ai-chat-messages');
            const msgDiv = document.createElement('div');
            
            if (role === 'user') {
                msgDiv.className = 'self-end bg-gray-900 text-white p-3 rounded-2xl rounded-tr-sm shadow-sm max-w-[85%] leading-relaxed';
                msgDiv.textContent = content;
            } else {
                msgDiv.className = 'self-start bg-white border border-gray-200 p-3 rounded-2xl rounded-tl-sm shadow-sm text-gray-700 max-w-[85%] leading-relaxed ai-markdown-content';
                msgDiv.innerHTML = typeof marked !== 'undefined' ? marked.parse(content) : content;
            }
            
            messagesDiv.appendChild(msgDiv);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
            return msgDiv;
        }

        async function sendAIChatMessage() {
            const inputEl = document.getElementById('ai-chat-input');
            const message = inputEl.value.trim();
            if (!message) return;

            appendMessage('user', message);
            inputEl.value = '';
            inputEl.disabled = true;

            const loadingMsg = appendMessage('assistant', '<i class="fa-solid fa-circle-notch fa-spin text-[#d46c75]"></i> 思考中...');

            const systemPrompt = `你現在是簡廉靜（毒防中心督導、PMP專案管理師、熱愛潛水/滑雪/攀岩/溯溪等極限運動與藍染/皮雕藝術手作的社工，喜歡打電玩和玩桌遊）的專屬 AI 履歷助理。
請以專業、友善且精煉的語氣（繁體中文），回答訪客關於簡廉靜背景、技能或經歷的問題。
每次回答請盡量控制在 100 字以內，適當使用 Emoji，並善用 Markdown 格式標示重點。`;

            try {
                const response = await fetchWithRetry(
                    `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`,
                    {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            contents: [{ parts: [{ text: message }] }],
                            systemInstruction: { parts: [{ text: systemPrompt }] }
                        })
                    }
                );

                const text = response.candidates?.[0]?.content?.parts?.[0]?.text;
                
                if (text) {
                    loadingMsg.innerHTML = typeof marked !== 'undefined' ? marked.parse(text) : text;
                } else {
                    throw new Error("無法取得 AI 回應");
                }
            } catch (error) {
                console.error(error);
                loadingMsg.innerHTML = '<i class="fa-solid fa-triangle-exclamation text-red-500 mr-1"></i> 哎呀，連線似乎有點問題，請稍後再試！';
            } finally {
                inputEl.disabled = false;
                inputEl.focus();
                const messagesDiv = document.getElementById('ai-chat-messages');
                messagesDiv.scrollTop = messagesDiv.scrollHeight;
            }
        }
